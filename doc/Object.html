def invalid_argv?
  ret = false
  month = ARGV[0].to_i
  day = ARGV[1].to_i
  
  unless 1 <= month || month <= 12
    ret = true
  end

  case month
  when 4, 6, 9, 11
    ret = !(1 <= day || day <= 30)
  when 2
    ret = !(1 <= day || day <= 29)
  else
    ret = !(1 <= day || day <= 31)
  end

  return ret
end

def life_clock(i, j, k, m)
  a = i
  b = j
  c = k
  n = m

  birth_time = Time.local(a, b, c).to_f
  death_time = Time.local(a + n + 1, b, c).to_f - 1.0
  now = Time.now.to_f
  a_sec_of_life = (death_time - birth_time) / (24 * 60 * 60).to_f

  life_clock_to_string(sec2hour((now - birth_time) / a_sec_of_life))
end

def sec2hour(s)
  hours = 0
  minites = 0
  
  while s >= 60.0
    if minites >= 60
      hours += 1
      minites = 0
    end
    s -= 60.0
    minites += 1
  end
  
  return [hours, minites, s]
end
